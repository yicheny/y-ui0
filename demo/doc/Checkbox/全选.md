- 通过`values`可以实时设置选项

### 示例

<!--start-code-->

```jsx
const allData = [1,2,3,4];

function Demo(){
    const [data,setData] = useState([]);
    return <div>
        <Checkbox onChange={(c)=>setData(c ? allData : [])}>All Check</Checkbox>
        <CheckboxGroup values={data} onChange={console.log}>
            <Checkbox value={1}>A</Checkbox>
            <Checkbox value={2}>B</Checkbox>
            <Checkbox value={3}>C</Checkbox>
            <Checkbox value={4}>D</Checkbox>
        </CheckboxGroup>
    </div>
}

ReactDOM.render(<Demo/>);
```
